// ── 靈感庫 — 已分類整理的靈感彙整 ──────────────────────────────
const R = dv.container;
const now = new Date();

// 讀取 Inbox
const inboxFile = app.vault.getAbstractFileByPath('07 Quick Capture & Hub/每日碎片化靈感隨記.md');
if (!inboxFile) { R.createEl('p', { text: '找不到 Inbox 檔案' }); return; }
const raw = await app.vault.read(inboxFile);

// 解析已分類的條目（有 [tag] 的行）
const entries = [];
for (const line of raw.split('\n')) {
  const t = line.trim();
  if (!/^[-*] /.test(t)) continue;
  const m = t.match(/^[-*] (✅)?\[([^\]]+)\]\s+(\d{4}-\d{2}-\d{2})\s+(\d{2}:\d{2})\s+([\s\S]+)/);
  if (!m) continue;
  entries.push({
    done:     !!m[1],
    category: m[2],
    date:     m[3],
    time:     m[4],
    content:  m[5].trim(),
  });
}

// 分類順序
const CAT_ORDER = ['發文素材', '生活計畫', '靈感', '雜念', '待辦'];
const grouped = {};
for (const e of entries) {
  const cat = CAT_ORDER.includes(e.category) ? e.category : '其他';
  if (!grouped[cat]) grouped[cat] = [];
  grouped[cat].push(e);
}

// 標題
const hdr = R.createEl('div', { attr: { style: 'padding:16px 0 8px;border-bottom:1px solid var(--db-border);margin-bottom:16px;' } });
hdr.createEl('h1', { attr: { style: 'font-size:20px;font-weight:700;margin:0 0 4px;' }, text: '💡 靈感庫' });
hdr.createEl('div', { attr: { style: 'font-size:12px;color:var(--db-text-muted);' }, text: `共 ${entries.length} 則 · ${now.getFullYear()}/${now.getMonth()+1}/${now.getDate()} 更新` });

const CAT_ICON = { '發文素材':'🎨', '生活計畫':'🌱', '靈感':'💡', '雜念':'💭', '待辦':'✅', '其他':'📌' };
const CAT_COLOR = { '發文素材':'var(--db-rose)', '生活計畫':'var(--db-teal)', '靈感':'var(--db-cream)', '雜念':'var(--db-text-muted)', '待辦':'var(--db-purple)', '其他':'var(--db-text-secondary)' };

const allCats = [...CAT_ORDER, '其他'].filter(c => grouped[c]?.length);

for (const cat of allCats) {
  const items = grouped[cat];
  const sec = R.createEl('div', { attr: { style: 'margin-bottom:24px;' } });

  // 分類標題
  const catHdr = sec.createEl('div', { attr: { style: 'display:flex;align-items:center;gap:8px;margin-bottom:10px;' } });
  catHdr.createEl('span', { attr: { style: `font-size:13px;font-weight:700;color:${CAT_COLOR[cat]};` }, text: `${CAT_ICON[cat]} ${cat}` });
  catHdr.createEl('span', { attr: { style: 'font-size:11px;color:var(--db-text-muted);background:var(--db-bg-deep);padding:1px 7px;border-radius:10px;' }, text: String(items.length) });

  for (const e of [...items].reverse()) { // 最新在前
    const card = sec.createEl('div', { attr: { style: 'background:var(--db-bg-card);border:1px solid var(--db-border);border-radius:10px;padding:12px 14px;margin-bottom:8px;' } });

    // 日期列
    const meta = card.createEl('div', { attr: { style: 'display:flex;align-items:center;gap:8px;margin-bottom:6px;' } });
    meta.createEl('span', { attr: { style: 'font-size:11px;color:var(--db-text-muted);' }, text: `${e.date} ${e.time}` });
    if (e.done) meta.createEl('span', { attr: { style: 'font-size:10px;color:var(--db-teal);' }, text: '✅ 已整理' });

    // 內容
    card.createEl('div', { attr: { style: 'font-size:13px;line-height:1.7;white-space:pre-wrap;word-break:break-word;' }, text: e.content });
  }
}

if (entries.length === 0) {
  R.createEl('div', { attr: { style: 'text-align:center;color:var(--db-text-muted);padding:40px 0;font-size:14px;' }, text: '目前尚無已分類的靈感' });
}
