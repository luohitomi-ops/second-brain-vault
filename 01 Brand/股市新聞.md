// ── 股市新聞 — 純閱讀，不進發文靈感池 ─────────────────────────
// 資料來源同 Brief（Gemini 自動生成），但股市新聞只用來看盤前資訊，
// 不算發文素材，跟 01 Brand/發文靈感池.md 分開顯示。
// 2026-07-10：改成可收合卡片（預設收合，只留每日總結常駐顯示），來源新聞點開才看。
const R = dv.container;

const hdr = R.createEl('div', { attr: { style: 'padding:16px 0 8px; border-bottom:1px solid var(--db-border); margin-bottom:16px;' } });
hdr.createEl('h1', { attr: { style: 'font-size:20px; font-weight:700; margin:0 0 4px;' }, text: '📈 股市新聞' });

const briefs = dv.pages('"01 Brand"')
  .where(p => p.type === 'brief')
  .sort(p => p.date, 'desc')
  .limit(14);

hdr.createEl('div', { attr: { style: 'font-size:12px; color:var(--db-text-muted);' }, text: `最近 ${briefs.length} 天 Brief 的股市新聞（每日 02:00 自動生成，僅供閱讀）` });

// 把 markdown 粗體/連結轉成安全的 HTML
function mdToHtml(text) {
  return text
    .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
    .replace(/\*\*(.+?)\*\*/g, '<b>$1</b>')
    .replace(/(https?:\/\/[^\s<]+)/g, '<a href="$1" target="_blank" style="color:var(--db-teal);">$1</a>');
}

if (briefs.length === 0) {
  R.createEl('div', { attr: { style: 'text-align:center; color:var(--db-text-muted); padding:40px 0; font-size:14px;' }, text: '尚無 Brief' });
} else {
  for (const b of briefs) {
    const raw = await dv.io.load(b.file.path);

    const summaryM = raw.match(/##\s*📌\s*股市今日重點總結\s*\n([\s\S]*?)(?=\n##|\n---|\s*$)/);
    const summary = summaryM ? summaryM[1].trim() : '';

    const newsM = raw.match(/##\s*📈\s*股市重要新聞\s*\n([\s\S]*?)(?=\n##|\n---|\s*$)/);
    const news = newsM ? newsM[1].trim() : '';

    const hasSummary = summary && !summary.includes('今日無股市重點總結') && !summary.includes('今日無可靠來源');
    const hasNews = news && !news.includes('今日無可靠來源');
    if (!hasSummary && !hasNews) continue; // 當天完全沒股市資訊就跳過，不佔版面

    const card = R.createEl('div', { attr: { style: 'background:var(--db-bg-card); border:1px solid var(--db-border); border-radius:10px; padding:14px 16px; margin-bottom:12px;' } });
    card.createEl('div', { attr: { style: 'font-size:12px; color:var(--db-purple); font-weight:600; margin-bottom:8px;' }, text: b.date || b.file.name });

    // 每日重點總結：常駐顯示，不用點開
    if (hasSummary) {
      const summaryEl = card.createEl('div', { attr: { style: 'font-size:13px; line-height:1.8; color:var(--db-text); margin-bottom:10px;' } });
      summaryEl.innerHTML = mdToHtml(summary);
    }

    // 來源新聞：預設收合，點標題展開
    if (hasNews) {
      const toggle = card.createEl('div', {
        attr: { style: 'font-size:12px; color:var(--db-teal); cursor:pointer; user-select:none; display:flex; align-items:center; gap:4px;' },
      });
      const chev = toggle.createEl('span', { text: '▸' });
      toggle.createEl('span', { text: '來源新聞（點開看細節）' });

      const body = card.createEl('div', { attr: { style: 'display:none; margin-top:10px; padding-top:10px; border-top:1px solid var(--db-border); font-size:13px; line-height:1.8; word-break:break-word;' } });
      body.innerHTML = mdToHtml(news).replace(/\n/g, '<br>');

      toggle.addEventListener('click', () => {
        const open = body.style.display !== 'none';
        body.style.display = open ? 'none' : 'block';
        chev.textContent = open ? '▸' : '▾';
      });
    }
  }
}
