```dataviewjs
// ── 發文靈感池 — Inbox 發文素材 + 每日 Brief ─────────────────────
const R = dv.container;

// ── 標題 ─────────────────────────────────────────────────────────
const hdr = R.createEl('div', { attr: { style: 'padding:16px 0 8px; border-bottom:1px solid var(--db-border); margin-bottom:16px;' } });
hdr.createEl('h1', { attr: { style: 'font-size:20px; font-weight:700; margin:0 0 4px;' }, text: '🎨 發文靈感池' });

// ── Section 1：每日 Brief（type: brief）──────────────────────────
const briefSection = R.createEl('div', { attr: { style: 'margin-bottom:24px;' } });
briefSection.createEl('div', { attr: { style: 'font-size:12px; font-weight:600; color:var(--db-text-muted); text-transform:uppercase; letter-spacing:.05em; margin-bottom:10px;' }, text: '📰 每日 Brief（社群熱點／熱門話題／時事梗，股市新聞另見側欄「股市新聞」不算靈感）' });

const briefs = dv.pages('"01 Brand"')
  .where(p => p.type === 'brief')
  .sort(p => p.date, 'desc')
  .limit(7);

if (briefs.length === 0) {
  briefSection.createEl('div', { attr: { style: 'font-size:13px; color:var(--db-text-muted); padding:8px 0;' }, text: '尚無 Brief，凌晨 02:00 自動生成' });
} else {
  for (const b of briefs) {
    const card = briefSection.createEl('div', { attr: { style: 'background:var(--db-bg-card); border:1px solid var(--db-border); border-radius:10px; padding:12px 14px; margin-bottom:8px; cursor:pointer;' } });
    card.addEventListener('click', () => app.workspace.getLeaf(false).openFile(app.vault.getAbstractFileByPath(b.file.path)));

    const meta = card.createEl('div', { attr: { style: 'display:flex; align-items:center; gap:8px; margin-bottom:4px;' } });
    meta.createEl('span', { attr: { style: 'font-size:11px; color:var(--db-text-muted);' }, text: b.date || b.file.name });
    const used = String(b['已使用'] || '⬜');
    const usedColor = used.includes('✅') ? 'var(--db-teal)' : 'var(--db-text-muted)';
    meta.createEl('span', { attr: { style: `font-size:10px; color:${usedColor};` }, text: used.includes('✅') ? '✅ 已使用' : '⬜ 未使用' });

    const summary = String(b['話題摘要'] || '').slice(0, 80);
    card.createEl('div', { attr: { style: 'font-size:13px; line-height:1.6; color:var(--db-text);' }, text: summary || b.file.name });
  }
}

// ── Section 2：Inbox 發文素材 ─────────────────────────────────────
const inboxSection = R.createEl('div');
inboxSection.createEl('div', { attr: { style: 'font-size:12px; font-weight:600; color:var(--db-text-muted); text-transform:uppercase; letter-spacing:.05em; margin-bottom:10px;' }, text: '🗂️ 碎片靈感（/整理 後出現）' });

const inboxFile = app.vault.getAbstractFileByPath('07 Quick Capture & Hub/每日碎片化靈感隨記.md');
if (!inboxFile) {
  inboxSection.createEl('p', { text: '找不到 Inbox 檔案' });
} else {
  const raw = await app.vault.read(inboxFile);
  const entries = [];
  for (const line of raw.split('\n')) {
    const t = line.trim();
    if (!/^[-*] /.test(t)) continue;
    const m = t.match(/^[-*] (✅)?\[([^\]]+)\]\s+(\d{4}-\d{2}-\d{2})\s+(\d{2}:\d{2})\s+([\s\S]+)/);
    if (!m) continue;
    if (m[2] !== '發文素材') continue;
    entries.push({ done: !!m[1], date: m[3], time: m[4], content: m[5].trim() });
  }

  if (entries.length === 0) {
    inboxSection.createEl('div', { attr: { style: 'font-size:13px; color:var(--db-text-muted); padding:8px 0;' }, text: '尚無已分類的發文素材，打完碎片後跑 /整理 即可' });
  } else {
    for (const e of [...entries].reverse()) {
      const card = inboxSection.createEl('div', { attr: { style: 'background:var(--db-bg-card); border:1px solid var(--db-border); border-radius:10px; padding:12px 14px; margin-bottom:10px;' } });
      const meta = card.createEl('div', { attr: { style: 'display:flex; align-items:center; gap:8px; margin-bottom:6px;' } });
      meta.createEl('span', { attr: { style: 'font-size:11px; color:var(--db-text-muted);' }, text: `${e.date} ${e.time}` });
      if (e.done) meta.createEl('span', { attr: { style: 'font-size:10px; color:var(--db-teal);' }, text: '✅ 已整理' });
      card.createEl('div', { attr: { style: 'font-size:13px; line-height:1.7; white-space:pre-wrap; word-break:break-word;' }, text: e.content });
    }
  }
}

// ── footer count ─────────────────────────────────────────────────
hdr.createEl('div', { attr: { style: 'font-size:12px; color:var(--db-text-muted);' }, text: `Brief ${briefs.length} 篇 · 發文素材碎片從 Inbox 動態載入` });
```
