const R = dv.container;
function el(parent, tag, opts = {}) {
  const { cls, text, html, attr = {} } = opts;
  const node = parent.createEl(tag, { cls, text });
  if (html) node.innerHTML = html;
  Object.entries(attr).forEach(([k, v]) => node.setAttribute(k, v));
  return node;
}
function openFile(path) {
  const f = app.vault.getAbstractFileByPath(path);
  if (f) app.workspace.getLeaf(false).openFile(f);
}
function makeSection(parent, title, defaultOpen = false) {
  const card = parent.createEl('div', { cls: 'db-card', attr: { style: 'margin-bottom:12px;' } });
  const hdr  = card.createEl('div', { cls: 'db-card-header', attr: { style: 'cursor:pointer;user-select:none;' } });
  el(hdr, 'div', { cls: 'db-card-title', text: title });
  const chev = el(hdr, 'div', { attr: { style: 'font-size:13px;color:var(--db-text-muted);' }, text: defaultOpen ? '▾' : '▸' });
  const body = card.createEl('div', { attr: { style: `display:${defaultOpen ? 'block' : 'none'};padding-top:6px;` } });
  hdr.addEventListener('click', () => {
    const open = body.style.display !== 'none';
    body.style.display = open ? 'none' : 'block';
    chev.textContent   = open ? '▸' : '▾';
  });
  return body;
}

const wrap = R.createEl('div', { attr: { style: 'max-width:860px;margin:0 auto;padding:12px 0;' } });

{
  const hc = wrap.createEl('div', { cls: 'db-card', attr: { style: 'margin-bottom:16px;padding:20px 24px;' } });

  el(hc, 'div', { attr: { style: 'font-size:13px;color:var(--db-text-muted);' }, text: '同一檔股票的持倉／估值／模型／研究，聚合顯示。不搬移任何原始檔案，純讀取視圖。' });
}

// ════════════════════════════════════════════════════════════════
// 1. 讀取各資料來源（純讀取，不修改任何檔案）
// ════════════════════════════════════════════════════════════════

// 1a. 波段庫存（持倉表）
let holdings = [];
try {
  const f = app.vault.getAbstractFileByPath('03 Invest-Swing/波段庫存.md');
  if (f) {
    const c = await app.vault.read(f);
    const lines = c.split('\n');
    let inTable = false;
    for (const ln of lines) {
      if (/^\|\s*代號\s*\|/.test(ln)) { inTable = true; continue; }
      if (inTable && /^\|---/.test(ln)) continue;
      if (inTable && !/^\|\s*\d/.test(ln)) { inTable = false; continue; }
      if (inTable) {
        const cols = ln.split('|').map(s => s.trim()).filter(Boolean);
        if (cols.length >= 6) holdings.push({ code: cols[0], name: cols[1], shares: cols[2], cost: cols[3], price: cols[4], pnl: cols[5], status: cols[6]||'', note: cols[7]||'' });
      }
    }
  }
} catch(e) {}

// 1b. 觀察名單
let watchlist = [];
try {
  const f = app.vault.getAbstractFileByPath('03 Invest-Swing/觀察名單.md');
  if (f) {
    const c = await app.vault.read(f);
    for (const m of c.matchAll(/^-\s*(\d{4,6})\s+(\S+)(?:（([^）]*)）)?/gm)) {
      watchlist.push({ code: m[1], name: m[2], note: m[3] || '' });
    }
  }
} catch(e) {}

// 1c. 財報摘要（估值快照 + AI解讀，按代號抓對應段落）
let valuationMap = {};
try {
  const f = app.vault.getAbstractFileByPath('03 Invest-Swing/財報摘要.md');
  if (f) {
    const c = await app.vault.read(f);
    for (const m of c.matchAll(/\*\*(\d{4,6})\s+([^\*]+)\*\*\s*(?:（ETF）)?\s*(現價[^\n]*)/g)) {
      valuationMap[m[1]] = { name: m[2].trim(), snapshot: m[3].trim() };
    }
    for (const m of c.matchAll(/\*\*(\d{4,6})\s+[^\*]+\*\*\n([^\n]+)/g)) {
      if (valuationMap[m[1]]) valuationMap[m[1]].comment = m[2].trim();
    }
  }
} catch(e) {}

// 1d. 財務模型檔案（按檔名開頭代號比對）
const modelFiles = dv.pages('"03 Invest-Swing/財務模型"').array();
function findModel(code) {
  return modelFiles.find(p => p.file.name.startsWith(code));
}

// 1e. 市場研究檔案（按檔名代號比對）
const researchFiles = dv.pages('"03 Invest-Swing/市場研究"').array();
function findResearch(code) {
  return researchFiles.find(p => p.file.name.startsWith(code));
}

// ════════════════════════════════════════════════════════════════
// 2. 分兩組：持倉中 vs 觀察中（尚未買入）
// ════════════════════════════════════════════════════════════════
const heldCodes = new Map();      // 目前持有
const watchOnlyCodes = new Map(); // 觀察中，尚未買入（不在持倉表裡）

holdings.forEach(h => heldCodes.set(h.code, { name: h.name }));
watchlist.forEach(w => { if (!heldCodes.has(w.code)) watchOnlyCodes.set(w.code, { name: w.name, note: w.note }); });
Object.entries(valuationMap).forEach(([code, v]) => {
  if (heldCodes.has(code)) return;
  if (!watchOnlyCodes.has(code)) watchOnlyCodes.set(code, { name: v.name, note: '' });
});

// ════════════════════════════════════════════════════════════════
// 3. 逐檔渲染收合卡片（共用函式）
// ════════════════════════════════════════════════════════════════
function renderStockCard(parentBody, code, info, isHeld) {
  const holding = holdings.find(h => h.code === code);
  const val = valuationMap[code];
  const model = findModel(code);
  const research = findResearch(code);

  const box = parentBody.createEl('div', { attr: { style: 'border:1px solid var(--db-border);border-radius:var(--db-radius-inner);margin-bottom:8px;overflow:hidden;' } });
  const bHdr = box.createEl('div', { attr: { style: 'display:flex;align-items:center;justify-content:space-between;padding:10px 14px;cursor:pointer;background:var(--db-bg-deep);user-select:none;' } });
  const bLeft = bHdr.createEl('div', { attr: { style: 'display:flex;align-items:center;gap:10px;min-width:0;' } });
  el(bLeft, 'code', { attr: { style: 'font-size:13px;font-weight:700;color:var(--db-cream);' }, text: `${code} ${info.name}` });
  if (isHeld && holding) el(bLeft, 'span', { cls: 'db-badge', attr: { style: 'font-size:10px;' }, text: holding.status || '持有' });
  if (!isHeld) el(bLeft, 'span', { cls: 'db-badge', attr: { style: 'font-size:10px;background:var(--db-teal);' }, text: '觀察中' });
  const bChev = el(bHdr, 'span', { attr: { style: 'font-size:12px;color:var(--db-text-muted);flex-shrink:0;' }, text: '▸' });

  const bBody = box.createEl('div', { attr: { style: 'display:none;padding:12px 14px;font-size:12px;line-height:1.8;border-top:1px solid var(--db-border);' } });

  if (isHeld && holding) {
    el(bBody, 'div', { attr: { style: 'font-size:10px;color:var(--db-text-muted);text-transform:uppercase;letter-spacing:.08em;margin-bottom:4px;' }, text: '持倉' });
    el(bBody, 'div', { attr: { style: 'color:var(--db-text-secondary);' }, text: `${holding.shares} 股｜均價 ${holding.cost}｜現價 ${holding.price}｜未實現 ${holding.pnl}` });
    if (holding.note) el(bBody, 'div', { attr: { style: 'color:var(--db-text-muted);font-size:11px;margin-top:2px;' }, text: holding.note });
  } else {
    el(bBody, 'div', { attr: { style: 'color:var(--db-text-muted);font-size:11px;' }, text: '尚未買入 — 觀察名單追蹤中' });
    if (info.note) el(bBody, 'div', { attr: { style: 'color:var(--db-text-muted);font-size:11px;margin-top:2px;' }, text: info.note });
  }

  if (val) {
    el(bBody, 'div', { attr: { style: 'font-size:10px;color:var(--db-text-muted);text-transform:uppercase;letter-spacing:.08em;margin:10px 0 4px;' }, text: '最新估值快照' });
    el(bBody, 'div', { attr: { style: 'color:var(--db-text-secondary);' }, text: val.snapshot });
    if (val.comment) el(bBody, 'div', { attr: { style: 'color:var(--db-text-muted);font-size:11px;margin-top:2px;' }, text: val.comment });
  }

  const linkRow = bBody.createEl('div', { attr: { style: 'display:flex;gap:8px;margin-top:10px;flex-wrap:wrap;' } });
  if (model) {
    const b = linkRow.createEl('button', { cls: 'db-btn db-btn-ghost db-btn-sm', text: '🏗️ 財務模型' });
    b.addEventListener('click', () => openFile(model.file.path));
  }
  if (research) {
    const b = linkRow.createEl('button', { cls: 'db-btn db-btn-ghost db-btn-sm', text: '🔍 市場研究' });
    b.addEventListener('click', () => openFile(research.file.path));
  }
  if (!model && !research) {
    el(linkRow, 'span', { attr: { style: 'color:var(--db-text-muted);font-size:11px;' }, text: `尚無模型/研究記錄，用 /建模 ${code} 或 /研究 ${code} 建立` });
  }

  bHdr.addEventListener('click', () => {
    const open = bBody.style.display !== 'none';
    bBody.style.display = open ? 'none' : 'block';
    bChev.textContent   = open ? '▸' : '▾';
  });
}

// ── 持倉中 ──
const heldBody = makeSection(wrap, `📋 持倉中（${heldCodes.size} 檔）`, true);
if (heldCodes.size === 0) {
  el(heldBody, 'div', { attr: { style: 'color:var(--db-text-muted);font-size:12px;padding:8px 0;' }, text: '目前無持倉，見 波段庫存.md' });
} else {
  for (const [code, info] of heldCodes) renderStockCard(heldBody, code, info, true);
}

// ── 觀察中，尚未買入 ──
const watchBody = makeSection(wrap, `🔭 觀察中，尚未買入（${watchOnlyCodes.size} 檔）`, true);
if (watchOnlyCodes.size === 0) {
  el(watchBody, 'div', { attr: { style: 'color:var(--db-text-muted);font-size:12px;padding:8px 0;' }, text: '目前觀察名單是空的，見 觀察名單.md（`- 代號 名稱（備註）` 格式新增）' });
} else {
  for (const [code, info] of watchOnlyCodes) renderStockCard(watchBody, code, info, false);
}
